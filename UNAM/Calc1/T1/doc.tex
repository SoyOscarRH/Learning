  Veamos la paridad
  \begin{itemize}
    \item $f(x) = -x^3+ 5x - 2$
    \begin{itemize}
      \item Par: 
      \begin{align*}
        f(-x) 
          &= -(-x)^3 + 5(-x) - 2 \\
          &= x^3 + 5(-x) - 2    \\
          &= x^3 - 5x - 2   
      \end{align*}
      Por lo tanto no es par

      \item Impar: 
      \begin{align*}
        -f(x) 
          &= x^3 - 5x + 2   
      \end{align*}
      Por lo tanto no es impar
    \end{itemize}

    \item $f(x) = -x^4 -2x^2 + 4$
    \begin{itemize}
      \item Par: 
      \begin{align*}
        f(-x) 
          &= -(-x)^4 + -2(-x)^2 + 4 \\
          &= -x^4 + -2x^2 + 4    \\
      \end{align*}
      Por lo tanto es par

      \item Impar: 
      \begin{align*}
        -f(x) 
          &= x^4 +2x^2 - 4 \\
      \end{align*}
      Por lo tanto no es impar
    \end{itemize}

    \item $f(x) = x\sqrt{x^2 - 1}$
    \begin{itemize}
      \item Par:

        f(-x)

      \end{align*}
      Por lo tanto no es par

      \item Impar: 
      \begin{align*}
        -f(x) 
          &= -x\sqrt{x^2 - 1} \\
      \end{align*}
      Por lo tanto no es impar

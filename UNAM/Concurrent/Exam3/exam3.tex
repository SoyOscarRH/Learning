    \begin{tcolorbox}
\begin{lstlisting}[language = JAVA]
public void enq(T x) {
  if (tail - head == capacity)
    throw FullException();
  items[tail % capacity] = x;
  tail++;
}

public T deq() {
  if (tail - head == 0)
    throw EmptyException();
  T x = items[head % capacity];
  head++;
  return x
}

\end{lstlisting}
